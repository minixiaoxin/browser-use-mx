## 方案1：使用自定义工具（推荐）

### 💡 原理
创建一个"等待确认"的工具，Agent 在关键步骤时必须调用这个工具，暂停等待人工确认。

### ✅ 优点
- 完全控制执行流程
- 可以在任何步骤暂停
- 支持多次确认
- 可以修改内容后继续

---

### 📝 实现代码

#### 基础版本

```python
from browser_use import Agent, Browser, Tools, ChatBrowserUse, ActionResult
from dotenv import load_dotenv
import asyncio

load_dotenv()

# 创建自定义工具
tools = Tools()

@tools.action('等待用户确认后再继续')
def wait_for_confirmation(action_description: str) -> ActionResult:
    """
    暂停执行，等待用户确认
    
    Args:
        action_description: 要执行的操作描述
    
    Returns:
        用户的确认结果
    """
    print(f"\n{'='*60}")
    print(f"⏸️  任务暂停 - 等待确认")
    print(f"{'='*60}")
    print(f"\n📝 即将执行的操作：")
    print(f"   {action_description}")
    print(f"\n💡 请在浏览器中检查内容")
    
    while True:
        choice = input("\n请选择操作 (y=继续, n=取消, e=修改): ").strip().lower()
        
        if choice == 'y':
            print("✅ 用户确认，继续执行...")
            return ActionResult(
                extracted_content="用户已确认，可以继续执行",
                success=True
            )
        elif choice == 'n':
            print("❌ 用户取消操作")
            return ActionResult(
                extracted_content="用户取消了操作",
                error="用户取消",
                success=False
            )
        elif choice == 'e':
            print("✏️ 请在浏览器中修改内容...")
            input("修改完成后按 Enter 继续...")
            print("✅ 修改完成，继续执行...")
            return ActionResult(
                extracted_content="用户已修改内容并确认",
                success=True
            )
        else:
            print("❌ 无效的选项，请输入 y/n/e")


async def main():
    browser = Browser(
        headless=False,  # 必须显示窗口
    )
    
    agent = Agent(
        task='''
        1. 访问某个表单页面
        2. 填写表单内容
        3. 填写完成后，使用 wait_for_confirmation 工具
        4. 等待用户确认
        5. 如果用户确认，则提交表单
        6. 如果用户取消，则停止任务
        ''',
        llm=ChatBrowserUse(),
        browser=browser,
        tools=tools,
    )
    
    history = await agent.run(max_steps=20)
    print(f"\n✅ 任务完成！")

if __name__ == "__main__":
    asyncio.run(main())
```

#### 增强版本（支持多次确认）

```python
from browser_use import Agent, Browser, Tools, ChatBrowserUse, ActionResult
from dotenv import load_dotenv
import asyncio

load_dotenv()

tools = Tools()

@tools.action('等待用户确认表单内容')
def confirm_form_content(form_description: str) -> ActionResult:
    """
    确认表单内容是否正确
    
    Args:
        form_description: 表单内容描述
    """
    print(f"\n{'='*60}")
    print(f"📋 表单内容确认")
    print(f"{'='*60}")
    print(f"\n{form_description}")
    print(f"\n💡 请在浏览器中检查表单内容")
    
    while True:
        choice = input("\n内容是否正确？(y=正确/n=需要修改): ").strip().lower()
        
        if choice == 'y':
            return ActionResult(
                extracted_content="表单内容已确认正确",
                success=True
            )
        elif choice == 'n':
            field = input("请输入需要修改的字段名: ").strip()
            new_value = input(f"请输入 {field} 的新值: ").strip()
            return ActionResult(
                extracted_content=f"需要修改：{field} = {new_value}",
                success=False
            )
        else:
            print("❌ 请输入 y 或 n")


@tools.action('等待用户确认提交')
def confirm_submit() -> ActionResult:
    """
    最终确认是否提交表单
    """
    print(f"\n{'='*60}")
    print(f"⚠️  最终确认")
    print(f"{'='*60}")
    print(f"\n即将提交表单！")
    print(f"请最后检查一次所有内容")
    
    choice = input("\n确认提交？(yes/no): ").strip().lower()
    
    if choice == 'yes':
        print("✅ 确认提交")
        return ActionResult(
            extracted_content="用户确认提交",
            success=True
        )
    else:
        print("❌ 取消提交")
        return ActionResult(
            extracted_content="用户取消提交",
            error="用户取消",
            success=False
        )


async def main():
    browser = Browser(headless=False)
    
    agent = Agent(
        task='''
        1. 访问表单页面
        2. 填写所有字段
        3. 使用 confirm_form_content 工具让用户确认内容
        4. 如果用户要求修改，则修改对应字段
        5. 重复步骤3-4直到用户确认内容正确
        6. 使用 confirm_submit 工具进行最终确认
        7. 只有在用户输入 "yes" 后才能提交表单
        8. 如果用户取消，则停止任务，不要提交
        ''',
        llm=ChatBrowserUse(),
        browser=browser,
        tools=tools,
    )
    
    history = await agent.run(max_steps=30)

if __name__ == "__main__":
    asyncio.run(main())
```

## 方案2：限制可用工具

### 💡 原理
移除或禁用 Agent 的提交相关工具，让 Agent 无法自动提交。

### 📝 实现代码

```python
from browser_use import Agent, Browser, Tools, ChatBrowserUse
from dotenv import load_dotenv
import asyncio

load_dotenv()

async def main():
    # 创建工具集，排除点击功能
    tools = Tools(
        exclude_actions=['click']  # 禁用点击，防止点击提交按钮
    )
    
    browser = Browser(headless=False)
    
    agent = Agent(
        task='''
        1. 访问表单页面
        2. 填写所有字段
        3. 填写完成后停止，不要提交
        4. 返回填写的内容供用户检查
        ''',
        llm=ChatBrowserUse(),
        browser=browser,
        tools=tools,
    )
    
    history = await agent.run(max_steps=15)
    
    print(f"\n✅ 表单已填写完成")
    print(f"📄 填写内容：{history.final_result()}")
    print(f"\n💡 请在浏览器中检查内容，手动提交")
    input("检查完成后按 Enter 关闭...")

if __name__ == "__main__":
    asyncio.run(main())
```

## 方案4：使用回调函数

### 💡 原理
监听 Agent 的每个动作，在关键操作前拦截并请求确认。

### 📝 实现代码

```python
from browser_use import Agent, Browser, ChatBrowserUse
from browser_use.agent.views import ActionResult
from dotenv import load_dotenv
import asyncio

load_dotenv()

# 需要确认的关键词
CONFIRM_KEYWORDS = ['submit', 'confirm', '提交', '确认', '完成', 'finish']

def should_confirm_action(action_name: str, params: dict) -> bool:
    """判断是否需要确认"""
    # 检查动作名称
    if any(keyword in action_name.lower() for keyword in CONFIRM_KEYWORDS):
        return True
    
    # 检查参数中的文本
    for value in params.values():
        if isinstance(value, str):
            if any(keyword in value.lower() for keyword in CONFIRM_KEYWORDS):
                return True
    
    return False


async def main():
    browser = Browser(headless=False)
    
    agent = Agent(
        task='''
        1. 访问表单页面
        2. 填写所有字段
        3. 提交表单
        ''',
        llm=ChatBrowserUse(),
        browser=browser,
    )
    
    # 注意：这是一个概念示例
    # 实际实现需要修改 Browser-Use 源码或使用自定义工具
    
    print("💡 提示：使用自定义工具方案更简单可靠")

if __name__ == "__main__":
    asyncio.run(main())
```

### 示例2：多步骤确认任务

```python
from browser_use import Agent, Browser, Tools, ChatBrowserUse, ActionResult
from dotenv import load_dotenv
import asyncio

load_dotenv()
```

## 最佳实践

### ✅ 推荐做法

1. **使用自定义确认工具**
   ```python
   @tools.action('等待确认')
   def wait_confirm(desc: str) -> ActionResult:
       # 实现确认逻辑
       pass
   ```

2. **在任务描述中明确说明**
   ```yaml
   task: |
     1. 填写表单
     2. 使用 wait_confirm 工具
     3. 只有确认后才能提交
   ```

3. **提供清晰的提示**
   ```python
   print("⏸️  任务暂停")
   print("📝 请检查内容")
   print("💡 确认后输入 y")
   ```

4. **支持多种选择**
   ```python
   # y=继续, n=取消, e=修改
   choice = input("(y/n/e): ")
   ```

---

### ❌ 避免的做法

1. **只依赖任务描述**
   - Agent 可能不遵守
   - 不够可靠

2. **没有明确的确认机制**
   - 容易误操作
   - 难以控制

3. **确认提示不清晰**
   - 用户不知道要做什么
   - 容易出错
